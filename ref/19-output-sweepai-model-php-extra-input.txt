<?php

namespace App\Services;

class OrderService
{
    private PriceCalculator $calculator;

    public function __construct(PriceCalculator $calculator)
    {
        $this->calculator = $calculator;
    }

    public function calculateTotal(array $items, float $discountPercent = 0): float
    {
        $subtotal = $this->calculator->calculateSubtotal($items);
        $discountedSubtotal = $this->calculator->applyDiscount($subtotal, $discountPercent);
        $tax = $this->calculator->calculateTax($discountedSubtotal);
        return $discountedSubtotal + $tax;
    }
}

<?php

namespace App\Cache;

use App\Contracts\CacheInterface;
use App\Support\Serializer;

class FileCache implements CacheInterface
{
    private string $dir;
    private Serializer $serializer;

    public function __construct(string $dir, Serializer $serializer)
    {
        $this->dir = $dir;
        $this->serializer = $serializer;
    }

    public function get(string $key, mixed $default = null): mixed
    {
        $file = $this->dir . '/' . $key;
        if (file_exists($file)) {
            return $this->serializer->unserialize(file_get_contents($file));
        }
        return $default;
    }

    public function set(string $key, mixed $value, int $ttl = 3600): bool
    {
        $file = $this->dir . '/' . $key;
        file_put_contents($file, $this->serializer->serialize($value));
        return true;
    }

    public function has(string $key): bool
    {
        return file_exists($this->dir . '/' . $key);
    }

    public function delete(string $key): bool
    {
        return unlink($this->dir . '/' . $key);
    }

    public function clear(): bool
    {
        return rmdir($this->dir);
    }
}

<?php

namespace App\Notifications;

use App\Mail\Mailer;
use App\Templating\TemplateEngine;
use App\Repositories\UserRepository;

class NotificationService
{
    private Mailer $mailer;
    private TemplateEngine $templates;
    private UserRepository $users;

    public function __construct(Mailer $mailer, TemplateEngine $templates, UserRepository $users)
    {
        $this->mailer = $mailer;
        $this->templates = $templates;
        $this->users = $users;
    }

    public function notifyUser(int $userId, string $message): bool
    {
        $user = $this->users->find($userId);
        if ($user === null) {
            return false;
        }
        $template = $this->templates->render('user-notification', $user);
        return $this->mailer->send($user['email'], "Notification", $template);
    }
}
